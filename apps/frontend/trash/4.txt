  const isMobile = useMediaQuery(theme.breakpoints.down("md"));
  const isDark = useMemo(() => theme.palette.mode === "dark", [theme.palette.mode]);
  const reduceMotion = useMediaQuery("(prefers-reduced-motion: reduce)");

  const moonRef = useParallax(0.12);

  const showEffects = useMemo(
    () => isDark && !reduceMotion,
    [isDark, reduceMotion]
  );

  const sections = useMemo(() => INFO.homepage.sections, []);

  const sectionHeaders = useMemo(
    () => [
      { label: "Our Experience", index: 1 },
      { label: "Our Workflows", index: 2 },
      { label: "Our Services", index: 3 },
    ],
    []
  );

  const heroSectionKey = useMemo(
    () => `hero-${isMobile}-${sections[0].title}`,
    [isMobile, sections]
  );

  return (
    <>
      {showEffects && (
        <>
          <Moon moonRef={moonRef} />
          <Stars />
        </>
      )}

      <GradientOverlay />

      <HeroSection 
        key={heroSectionKey}
        isMobile={isMobile} 
        sections={sections} 
      />

      {showEffects && <GlowEffects position={["bottom-left", "bottom-right"]} />}

      <SectionContent
        showEffects={showEffects}
        sectionHeaders={sectionHeaders}
        sections={sections}
      />
    </>
  );
};
